var arenaRecord = std.heap.ArenaAllocator.init(std.heap.page_allocator);

    arenaRecord.deinit();
    arenaRecord = std.heap.ArenaAllocator.init(std.heap.page_allocator);
    allocatorRecord = arenaRecord.allocator();
}
